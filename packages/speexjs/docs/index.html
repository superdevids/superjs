Overview

v0.7.1

SpeexJS is a full-stack JavaScript/TypeScript framework for building modern web applications. It provides a cohesive development experience with server-side routing, middleware, database ORM, authentication, caching, file storage, real-time communication, and a reactive client — all under a single, zero-dependency core.

Designed for speed and simplicity, SpeexJS follows a Laravel-inspired architecture while embracing modern ECMAScript standards and TypeScript first-class support.

Key Features

  • Full-stack TypeScript with ESM native support
  • Laravel-style routing, controllers, and middleware pipeline
  • Multi-dialect database ORM (MySQL, SQLite, PostgreSQL)
  • Built-in authentication, caching, and file storage
  • Reactive client with signals and VDOM (JSX support)
  • Type-safe RPC layer
  • Schema validation
  • Zero runtime dependencies
  • CLI for scaffolding and code generation

Quick Start

Create a new project

npx speexjs init my-app
# or
speexjs init my-app

Start the development server

cd my-app
speexjs serve

Create a controller

speexjs make:controller UserController

Define a route

import { speexjs } from 'speexjs/server'

const app = speexjs()

app.get('/', ctx => {
  ctx.response.html('<h1>Hello SpeexJS</h1>')
})

app.listen(3000)

Installation

# Install globally
npm install -g speexjs

# Or use in your project
npm install speexjs

SpeexJS requires Node.js 18+. No other runtime dependencies are needed.

Server

The SuperApp class is the core of every SpeexJS application. It bootstraps routing, middleware, and lifecycle management.

import { speexjs } from 'speexjs/server'

const app = speexjs({ /* options */ })

  console.log('Running on port 3000')

Routing

SpeexJS provides a clean, expressive routing API inspired by Laravel.

Basic routes

app.get('/users', handler)
app.post('/users', handler)
app.put('/users/:id', handler)
app.patch('/users/:id', handler)
app.delete('/users/:id', handler)

// Match multiple methods
app.match(['GET', 'POST'], '/path', handler)

// Any method
app.any('/path', handler)

Route groups

app.group('/admin', router => {
  router.get('/dashboard', dashboardHandler)
  router.get('/users', adminUsersHandler)
})

Route parameters

app.get('/users/:id', ctx => {
  const { id } = ctx.params
  ctx.response.json({ userId: id })
})

Controllers

Controllers group related route handlers into a single class using decorators.

import { Controller, get, post } from 'speexjs/server'
import type { RouteContext } from 'speexjs/server/router'

@Controller('/users')
export class UserController {
  @get('/')
  async index(ctx: RouteContext) {
    ctx.response.json({ users: [] })
  }

  @post('/')
  async store(ctx: RouteContext) {
    const data = ctx.request.body
    ctx.response.status(201).json({ created: data })
  }
}

Resource controllers

app.resource('users', UserController)
// Generates: index, create, store, show, edit, update, destroy

app.apiResource('users', UserController)
// Generates: index, store, show, update, destroy

Container

The dependency injection container manages class and instance resolution for your application.

import { Container } from 'speexjs/server/container'

const container = new Container()

container.bind('mailer', SmtpMailTransport)
container.singleton('db', DatabaseConnection)
container.instance('config', { key: 'value' })

// Resolve from anywhere
const mailer = container.make('mailer')

Models

  protected table = 'users'

    return this.hasMany(Post)
  }
}

// Usage
const users = await User.query().where('active', true).get()

user.name = 'Updated'
await user.save()

Migrations

import { SchemaBuilder } from 'speexjs/server/database'

export const up = async (schema: SchemaBuilder) => {
  await schema.create('users', table => {
    table.id()
    table.string('name')
    table.string('email').unique()
    table.string('password')
    table.timestamps()
  })
}

export const down = async (schema: SchemaBuilder) => {
  await schema.drop('users')
}
# Run migrations
speexjs migrate

# Seed database
speexjs db:seed

Query Builder

The fluent query builder provides a clean interface for constructing SQL queries.

const users = await db
  .table('users')
  .select('id', 'name', 'email')
  .where('active', true)
  .where('role', 'admin')
  .orderBy('created_at', 'desc')
  .limit(20)
  .offset(0)
  .get()

// Aggregates
const count = await db.table('users').count()
const exists = await db.table('users').where('email', email).exists()

// Joins
const rows = await db
  .table('posts')
  .join('users', 'users.id', 'posts.user_id')
  .get()

Validation

import { schema } from 'speexjs/schema'

const userSchema = schema.object({
  name: schema.string().min(2).max(255),
  email: schema.string().email(),
  age: schema.number().min(18),
})

// In a controller
app.post('/users', validate(userSchema), ctx => {
  const data = ctx.request.body // Already validated
})

Authentication

SpeexJS provides built-in authentication with session and token-based strategies.

import { Auth } from 'speexjs/server/auth'

const auth = new Auth(db)

// Login
const user = await auth.attempt(email, password)

// Current user
const current = auth.user()

// Check authentication
if (auth.check()) {
  // User is logged in
}

Events

import { Event, event } from 'speexjs/server'

// Define an event
export class UserRegistered extends Event {
  constructor(public user: User) { super() }
}

// Listen
event.on(UserRegistered, async (evt) => {
  await sendWelcomeEmail(evt.user)
})

// Dispatch
await event.dispatch(new UserRegistered(user))

Queues

Queue jobs are processed asynchronously in the background, supporting delayed execution and retries.

import { Queue } from 'speexjs/server/queue'

const queue = new Queue()

// Dispatch a job
await queue.push('send-welcome-email', { userId: 1 })

// Delayed job
await queue.push('send-reminder', { userId: 1 }, { delay: 3600 })

Scheduling

import { Schedule } from 'speexjs/server/schedule'

const schedule = new Schedule()

// Run a task every minute
schedule.everyMinute(async () => {
  console.log('Running scheduled task...')
})

// Daily at midnight
schedule.daily('00:00', async () => {
  await cleanupOldRecords()
})

Testing

SpeexJS ships with testing utilities for HTTP requests, database refresh, and in-memory SQLite.

import { describe, it, expect } from 'vitest'
import { testRequest } from 'speexjs/server/testing'
import { RefreshDatabase } from 'speexjs/server/testing'
import { speexjs } from 'speexjs/server'

describe('Users API', () => {
  const app = speexjs()
  const http = testRequest(app)
  
  const refresh = new RefreshDatabase()
  const db = await RefreshDatabase.createSqliteMemory()
  refresh.setConnection(db).tables('users')

  beforeEach(async () => {
    await refresh.refresh()
  })

  it('lists users', async () => {
    const res = await http.get('/users')
    expect(res.status).toBe(200)
  })
})
